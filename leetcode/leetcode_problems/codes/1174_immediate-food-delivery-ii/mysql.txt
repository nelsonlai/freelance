WITH CustomerFirstOrder AS (
    SELECT 
        customer_id,
        MIN(order_date) AS first_order_date
    FROM Delivery
    GROUP BY customer_id
),
CustomerFirstOrderImmediate AS (
    SELECT 
        d.customer_id,
        CASE 
            WHEN d.order_date = d.customer_pref_delivery_date THEN 1 
            ELSE 0 
        END AS is_immediate
    FROM Delivery d
    JOIN CustomerFirstOrder cfo ON d.customer_id = cfo.customer_id 
        AND d.order_date = cfo.first_order_date
)
SELECT 
    ROUND(AVG(is_immediate) * 100, 2) AS immediate_percentage
FROM CustomerFirstOrderImmediate;
