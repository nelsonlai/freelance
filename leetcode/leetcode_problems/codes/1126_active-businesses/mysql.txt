SELECT 
    e.business_id
FROM Events e
JOIN (
    SELECT 
        event_type,
        AVG(occurences) AS avg_occurences
    FROM Events
    GROUP BY event_type
) avg_events ON e.event_type = avg_events.event_type
WHERE e.occurences > avg_events.avg_occurences
GROUP BY e.business_id
HAVING COUNT(DISTINCT e.event_type) > 1;
