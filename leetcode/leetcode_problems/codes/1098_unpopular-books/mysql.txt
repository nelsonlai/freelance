-- Problem: Unpopular Books
-- Difficulty: Medium
-- Tags: Database

SELECT b.book_id, b.name
FROM Books b
LEFT JOIN Orders o 
    ON b.book_id = o.book_id 
    AND o.dispatch_date >= '2018-06-23'
WHERE b.available_from < '2019-05-23'
GROUP BY b.book_id, b.name
HAVING IFNULL(SUM(o.quantity), 0) < 10;
