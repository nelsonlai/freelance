SELECT DISTINCT l1.id, a.name
FROM Logins l1
JOIN Accounts a ON l1.id = a.id
WHERE EXISTS (
    SELECT 1
    FROM Logins l2
    WHERE l2.id = l1.id
      AND l2.login_date = DATE_ADD(l1.login_date, INTERVAL 1 DAY)
)
AND EXISTS (
    SELECT 1
    FROM Logins l3
    WHERE l3.id = l1.id
      AND l3.login_date = DATE_ADD(l1.login_date, INTERVAL 2 DAY)
)
AND EXISTS (
    SELECT 1
    FROM Logins l4
    WHERE l4.id = l1.id
      AND l4.login_date = DATE_ADD(l1.login_date, INTERVAL 3 DAY)
)
AND EXISTS (
    SELECT 1
    FROM Logins l5
    WHERE l5.id = l1.id
      AND l5.login_date = DATE_ADD(l1.login_date, INTERVAL 4 DAY)
)
ORDER BY l1.id;
