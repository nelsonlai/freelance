📦 JavaScript 變數示範

1. let - 可變的變數

let name = '小明';
console.log(name); // 小明

name = '小華'; // 可以重新賦值
console.log(name); // 小華

2. const - 不可變的常數

const PI = 3.14159;
console.log(PI); // 3.14159

// PI = 3.14; // ❌ 錯誤！不能改變常數

3. 變數命名規則

// ✅ 正確的命名
let userName = '小明';
let userAge = 25;
let isStudent = true;
let $price = 100;
let _private = 'secret';

// ❌ 錯誤的命名
// let 2name = 'error'; // 不能以數字開頭
// let user-name = 'error'; // 不能用連字號
// let user name = 'error'; // 不能有空格

4. 變數作用域

let globalVar = '全域變數';

function testScope() {
  let localVar = '區域變數';
  console.log(globalVar); // 可以存取
  console.log(localVar); // 可以存取
}

// console.log(localVar); // ❌ 錯誤！無法存取
